-- Create fuel stocks table
CREATE TABLE fuel_stocks (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  shed_id UUID REFERENCES sheds(id) ON DELETE CASCADE,
  fuel_type TEXT NOT NULL,
  stock_liters FLOAT DEFAULT 0,
  price_rs FLOAT DEFAULT 0,
  last_updated TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()),
  UNIQUE(shed_id, fuel_type) -- Ensures one row per fuel type per shed
);

-- Enable RLS (Security)
ALTER TABLE fuel_stocks ENABLE ROW LEVEL SECURITY;

-- Allow sheds to update their own data
CREATE POLICY "Sheds can update their own stock" 
ON fuel_stocks FOR ALL 
USING (auth.uid() = shed_id);

ALTER TABLE fuel_stocks ADD COLUMN next_arrival DATE;
